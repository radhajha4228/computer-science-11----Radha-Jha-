Arrays in C


    
Definition

    
An array is a collection of elements of the same data type, stored in contiguous memory locations. It allows storing multiple values in a single variable.


    
Syntax

    
datatype arrayName[size];
    


    
Example:

    
#include <stdio.h>
int main() {
    int numbers[5] = {10, 20, 30, 40, 50};
    for (int i = 0; i < 5; i++) {
        printf("%d ", numbers[i]);
    }
    return 0;
}
    


    
Output:

    
10 20 30 40 50


    
1D Array

    
Definition

    
A one-dimensional array is a list of elements stored in a linear sequence.


    
Syntax:

    
datatype arrayName[size];
    


    
Example:

    
#include <stdio.h>
int main() {
    int arr[3] = {1, 2, 3};
    for (int i = 0; i < 3; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}
    


    
Output:

    
1 2 3


    
2D Array

    
Definition

    
A two-dimensional array is an array of arrays, forming a matrix-like structure.


    
Syntax:

    
datatype arrayName[rows][columns];
    


    
Example:

    
#include <stdio.h>
int main() {
    int matrix[2][2] = {{1, 2}, {3, 4}};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
    return 0;
}
    


    
Output:

    
1 2
3 4
    


    
3D Array

    
Definition

    
A three-dimensional array is an array of two-dimensional arrays, used to store multi-layered data.


    
Syntax:

    
datatype arrayName[size1][size2][size3];
    


    
Example:

    
#include <stdio.h>
int main() {
    int arr[2][2][2] = {{{1, 2}, {3, 4}}, {{5, 6}, {7, 8}}};
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            for (int k = 0; k < 2; k++) {
                printf("%d ", arr[i][j][k]);
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
    


    
Output:

    
1 2
3 4

5 6
7 8
